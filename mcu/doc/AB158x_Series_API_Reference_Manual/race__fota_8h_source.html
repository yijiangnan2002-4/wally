81 typedef enum {
82  RACE_FOTA_MODE_BACKGROUND = 0, /* FOTA background mode: FOTA download continues even if a call or music is on-going. FOTA download speed is low and will not be affected by call or music. */
83  RACE_FOTA_MODE_ACTIVE, /* FOTA active mode: A call has greater priority than the FOTA download which has priority over music. For example, when there is an on-going call, the FOTA download stops. When FOTA download starts, music playback will be paused. The download speed for FOTA is high. */
84  RACE_FOTA_MODE_ADAPTIVE, /* FOTA adaptive mode: FOTA download continues even if an ongoing call or music playback. However the download speed will decrease when there is call or music on-going. The FOTA download speed increases when there is neither call nor music. */
85 
86  RACE_FOTA_MODE_MAX /* The maximum value of this enum. It must not exceed 0x0F. */
87 } race_fota_mode_enum;
