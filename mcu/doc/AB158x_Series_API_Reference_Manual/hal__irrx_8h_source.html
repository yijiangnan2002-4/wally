213 /*
214 * - Use IRRX in the PWD mode. \n
215 * - Step1: Call #hal_gpio_init() to init the pins, if EPT tool hasn't been used to configure the related pinmux.
216 * - Step2: Set GPIO34 to RX function to enable the IRRX mode.
217 * - Step3: Call hal_irrx_init() to initialize the IRRX.
218 * - Step4: Set the PWD parameters: inverse and terminate_threshold.
219 * - Step5: Call hal_irrx_receive_pwd_start() to start the hardware.
220 * - Step6: Call hal_irrx_receive_pwd() to receive the PWD code.
221 * - Step7: Call hal_irrx_receive_pwd_start() once again to restart the PWD mode.
222 * - Step8: Call hal_irrx_deinit() to deinitialize the IRRX.
223 * - sample code:
224 * @code
225 * //Init GPIO, set GPIO pinmux(if EPT tool hasn't been used to configure the related pinmux).
226 * hal_gpio_init(HAL_GPIO_34);
227 *
228 * //Call hal_pinmux_set_function() to set GPIO pinmux, for more information, please reference hal_pinmux_define.h
229 * //function_index = HAL_GPIO_34_IR_RX.Need not to configure pinmux if EPT tool is used.
230 * hal_pinmux_set_function(HAL_GPIO_34, function_index);//Set the GPIO34 to IRRX mode.
231 *
232 * hal_irrx_init();
233 * uint32_t us = 0;
234 * hal_irrx_pwd_config_t format = {0}; //Initializes the PWD parameter.
235 * format.inverse = 1;
236 * format.terminate_threshold = 10200;
237 * if (HAL_IRRX_STATUS_OK != hal_irrx_receive_pwd_start(&format,
238 * user_callback,
239 * NULL,
240 * &us)) {
241 * dbg_error("hal_irrx_receive_pwd_start fail!");
242 * }
243 * if (HAL_IRRX_STATUS_OK != hal_irrx_receive_pwd(&received_length,
244 * data,
245 * sizeof(data))) {
246 * dbg_error("hal_irrx_receive_pwd fail!");
247 * }
248 * @endcode
249 * @code
250 * // Callback function. This function should be registered with #hal_irrx_receive_pwd().
251 * void user_callback (void *user_data)
252 * {
253 * hal_irrx_deinit();
254 * // user's handler
255 * }
256 *
257 * @endcode
258 *
259 *
260 * - Use IRRX driver to trigger IRRX RC5/PWD manually. \n
261 *
262 *
263 */
264 #endif
